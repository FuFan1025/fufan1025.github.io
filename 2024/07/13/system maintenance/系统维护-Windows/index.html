# 操作命令

# 查看当前计算机的初始安装日期和系统启动时间

systeminfo |findstr -i "初始安装日期 系统启动时间"

# 查看系统共享服务

net share #查看服务

net share 服务名 #开启服务共享

net share 服务名 /delete #删除开启的共享服务

# 关机指令

  • /s 关机指令
  • /t 设置关机时间 例：3600 为一个小时
  • /a 取消关机
shutdown /s /t "times"

# 生成目录树结构

tree "文件夹路径" > xxx.txt

# 拷贝文件

copy text.txt c:\doc\key_bak.txt

# 查找文件中 ok

find "ok" '文件路径' && echo "成功" || echo "不成功"

# 查看网络连接以及端口

netstat -ano

# 将硬盘分区从 FAT32 转换为 NTFS

convert f: /fs:ntfs /x #f为要转换的盘符

# 通过命令跟踪 TCP/IP 数据包路径

tracert www.baidu.com

# 打开本地 Windows 服务界面

services.msc

# 打开本地安全策略窗口

secpol.msc

# 打开屏幕键盘

OSK

# 打开注册表

regedit

# 操作

# 注册表查询开机自启动项目列表

计算机 \HKEY_CURRENT_USER\Software\Microsoft\Windows\CurrentVersion\Run

# 计算机配置【指定域名】和 192.168.0.138 IP 绑定

  1. 打开 C:\Windows\System32\drivers\etc\hosts
  2. 在文件的末尾添加一行，格式为 “IP 地址 域名”，例如 “192.168.0.138 指定域名”

# 配置任务计划创建磁盘碎片清理

  1. cmd 键入 taskschd.msc
  2. 应用程序选择 defrag.exe
